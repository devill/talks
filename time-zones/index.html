public function isTimeOfDayPastScheduleTime($now, $firstRun)
{
    if($this->IsScheduledToDST($now, $firstRun)) {
        return $this->getTodaysScheduleTime($now, $firstRun) <= $now + 3600;
    }

    return $this->getTodaysScheduleTime($now, $firstRun) <= $now;
}

private function IsScheduledToDST($now, $dateOfFirstRecurrence)
{
    $time = $this->getTodaysScheduleTime($now, $dateOfFirstRecurrence);
    return date('Y-m-d H:i:s', $time - 3600) == date('Y-m-d H:i:s', $time);
}
