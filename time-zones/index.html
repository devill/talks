$currentDay = time();

$isOnDST = date('Y-m-d H:i:s', $todaysSchedule - 3600) ==
           date('Y-m-d H:i:s', $todaysSchedule);

if ($isOnDST)
{
    $currentDay += 3600;
}

if (strtotime($todaysSchedule) <= $currentDay)
{
    $currentDay += 86400;
}

return date('Y-m-d', $currentDay).' '.substr($startDate, 11);
